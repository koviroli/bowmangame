1   import action.ShotAction;
2   import controller.ArcherController;
3   import controller.ArrowController;
4   import controller.BowController;
5   import controller.TargetController;
6   import javafx.application.Platform;
7   import javafx.event.EventHandler;
8   import javafx.scene.Group;
9   import javafx.scene.Scene;
10  import javafx.scene.input.MouseEvent;
11  import javafx.stage.Stage;
12  import model.BowManGameObject;
13  import view.GameUI;
14  
15  public class Game extends BowManGameObject{
16  	
17  	private static Group root = new Group();
18  	private static Scene scene = new Scene(root, WINDOW_WIDTH, WINDOW_HEIGHT);
19  	private Stage stage = new Stage();
20  	
21  	private ArcherController archController = new ArcherController();
22  	private BowController bowController = new BowController();
23  	private ArrowController arrowController = new ArrowController();
24  	private TargetController targetController = new TargetController();
25  	
26  	private GameUI gameUI = new GameUI();
27  	private ShotAction shot = new ShotAction(arrowController.getModel(), targetController.getModel());
28  	
29  	public Game(){
30  		stage.setScene(scene);
31  		addToRoot();
32  		eventHandler();
33  		stage.show();
34  		stage.setOnCloseRequest(e -> Platform.exit());
35  	}
36  	
37  	
38  	//handle keyboard-, mouse user input
39  	private void eventHandler(){
40  		scene.addEventHandler(MouseEvent.MOUSE_PRESSED, new EventHandler<MouseEvent>() {
41              public void handle(MouseEvent e) {            
42                  if (e.getClickCount() > 0) {         	
43                  	gameUI.setAimLineStartXY(e.getX(), e.getY());
44                  }  
45              }
46          });
47  		
48  		scene.addEventHandler(MouseEvent.MOUSE_DRAGGED, new EventHandler<MouseEvent>() {
49              public void handle(MouseEvent e) {            
50                  if (e.getClickCount() > 0) { 	
51                  	gameUI.setAimLineEndXY(e.getX(), e.getY());
52                  	gameUI.UpdateGameInfoTexts();
53                  	rotationHandler();
54                  	shot.setPower(gameUI.getPower());
55                  	shot.setAngle(gameUI.getAngle());
56                  }  
57              }
58          });
59  		
60  		scene.addEventHandler(MouseEvent.MOUSE_CLICKED, new EventHandler<MouseEvent>() {
61              public void handle(MouseEvent e) {            
62                  if (e.getClickCount() > 0) { 
63                  	gameUI.clearAimLine();
64  
65                      if( shot.action() ){
66                      	gameUI.incPoints();
67                      }
68                      else{
69                      	gameUI.decrPoints();
70                      }
71                  }  
72              }
73          });
74  	}
75  	
76  	//set the rotation of the models and also rotate them
77  	public void rotationHandler(){
78  		bowController.setRotation(gameUI.getAngle());
79  		arrowController.setRotation(gameUI.getAngle());
80  		bowController.rotate(archController.getModel().getCenterX(), archController.getModel().getCenterY());
81  		arrowController.rotate(archController.getModel().getCenterX(), archController.getModel().getCenterY());
82  	}
83  	
84  	//this function add elements to root
85  	private void addToRoot(){
86  		root.getChildren().add(archController.setView().getCanvas());
87  		root.getChildren().add(bowController.getView().getCanvas());
88  		root.getChildren().add(arrowController.getView().getCanvas());
89  		root.getChildren().add(targetController.getView().getCanvas());
90  		root.getChildren().add(gameUI.getAngleText());
91  		root.getChildren().add(gameUI.getPowerText());
92  		root.getChildren().add(gameUI.getAimLine());
93  		root.getChildren().add(gameUI.getPointsText());
94  		root.getChildren().add(shot.getImw());
95  		root.getChildren().add(shot.getPath());
96  	}
97  	
98  	//gives back the scene we builded
99  	public static Scene scene(){ 
100 		return scene; 
101 	}	
102 }
